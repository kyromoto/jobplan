API:

- /jobs
    - POST: Job anlegen
        - REQ: resource job ohne ID
        - RES: resource job
    - GET: Alle Jobs lesen
        - Parameter:
            - ?orderBy=<field_name>
            - ?orderDir=<asc><desc>
            - ?date_left (inklusive)
            - ?date_right (exklusive)
        - REQ: ---
        - RES: Liste mit allen Resourcen (Jobs)
- /jobs/<id>
    - GET: Job lesen
        - REQ: ---
        - RES: resource job
    - PATCH: Job ändern
        - REQ: resource job mit allen feldern die geändert werden sollen
        - RES: ???

        - REQ: ---
